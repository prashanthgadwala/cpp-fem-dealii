209 template <int dim>
210 void Solid<dim>::run()
211 {
212  make_grid();
213  system_setup();
214  //output initial values (here: =0)
215  output_results();
216  //Loop over the number of load_steps (see class declaration)
217  for (current_load_step=1; current_load_step <= load_steps; current_load_step++)
218  {
219  /*Always reset the increment vector - not to be mistaken with
220  the newton update!!!*/
221  solution_delta = 0.0;
222  /*Compute for the current load step the incremental solution using
223  Newton-Rapshon*/
224  solve_load_step_NR(solution_delta);
225  /*add the converged delta to the solution - not to be mistaken with
226  the newton update!!!*/
227  solution_n += solution_delta;
228  output_results();
229  }
230 }

482 template <int dim>
483 void Solid<dim>::assemble_system()
484 {
485 
486 
487  std::cout << " Assemble System " << std::flush;
488 
489 
490  NeoHookeanMaterial<dim> material(this->mu, this->lambda);
491 
492  //FEValues and FaceValues to compute quantities on quadrature points for our finite
493  //element space including mapping from the real cell
494  FEValues<dim> fe_values_ref (fe,//The used FiniteElement
495  qf_cell,//The quadrature rule for the cell
496  update_values| //UpdateFlag for shape function values
497  update_gradients| //shape function gradients
498  update_JxW_values); //transformed quadrature weights multiplied with Jacobian of transformation
499  FEFaceValues<dim> fe_face_values_ref (fe,
500  qf_face, //The quadrature for face quadrature points
501  update_values|
502  update_normal_vectors| //compute normal vector for face
503  update_JxW_values);
504 
505  //Quantities to store the local rhs and matrix contribution
506  FullMatrix<double> cell_matrix(dofs_per_cell,dofs_per_cell);
507  Vector<double> cell_rhs (dofs_per_cell);
508  //Vector with the indicies (global) of the local dofs
509  std::vector<types::global_dof_index> local_dof_indices (dofs_per_cell);
510  //Compute the current, total solution, i.e. starting value of
511  //current load step and current solution_delta
512  Vector<double> current_solution = get_total_solution(this->solution_delta);
513  //Iterators to loop over all active cells
514  typename DoFHandler<dim>::active_cell_iterator cell = dof_handler_ref.begin_active(),
515  endc = dof_handler_ref.end();
516 
517  for(;cell!=endc;++cell)
518  {
519  //Reset the local rhs and matrix for every cell
520  cell_matrix=0.0;
521  cell_rhs=0.0;
522  //Reinit the FEValues instance for the current cell, i.e.
523  //compute the values for the current cell
524  fe_values_ref.reinit(cell);
525  //Vector to store the gradients of the solution at
526  //n_q_points quadrature points
527  std::vector<Tensor<2,dim> > solution_grads_u(n_q_points);
528  //Fill the previous vector using get_function_gradients
529  fe_values_ref[u_fe].get_function_gradients(current_solution,solution_grads_u);
530  //Write the global indicies of the local dofs of the current cell
531  cell->get_dof_indices(local_dof_indices);
532 
533  //Loop over all quadrature points of the cell
534  for(unsigned int k=0; k<n_q_points;++k)
535  {
536  //BEGIN - INSERT YOUR CODE HERE
537  //Compute here the following
538  //
539  //- deformation gradient using the information in "solution_grads_u[k]" and Physics::Elasticity::StandardTensors<dim>::I
540  //- use the deformation gradient to compute
541  //- - Kirchhoffstress
542  //- - Tangent
543  //- Compute the inverse of the Deformation gradient
544  Tensor<2,dim> DeformationGradient = (Tensor<2, dim>(Physics::Elasticity::StandardTensors<dim>::I) + solution_grads_u[k]);
545  SymmetricTensor<2,dim> Kirchhoffstress = material.get_KirchhoffStress(DeformationGradient);
546  SymmetricTensor<4,dim> Tangent = material.get_Tangent_spt(DeformationGradient);
547  Tensor<2,dim> F_inv = invert(DeformationGradient);
548  //END - INSERT YOUR CODE HERE
549 
550  //The quadrature weight for the current quadrature point
551  const double JxW = fe_values_ref.JxW(k);
552  //Loop over all dof's of the cell
553  for(unsigned int i=0; i<dofs_per_cell; ++i)
554  {
555  //Assemble system_rhs contribution
556  //BEGIN - INSERT YOUR CODE HERE
557 
558  //Compute here the following
559  //
560  //- the symmetric (SymmetricTensor<2,dim>) gradient with respect to the spatial configuration
561  // for the test function as discussed in the exercise (i) -> use fe_values_ref[u_fe] and F_inv
562  //- write into cell_rhs(i)-= the contribution for the current test function at the current quadrature point
563  // !! "-=" due to Newton-Raphson algorithm K\du = -r
564  Tensor<2,dim> shape_gradient_wrt_ref_config_i = fe_values_ref[u_fe].gradient(i,k);
565  Tensor<2,dim> shape_gradient_wrt_spt_config_i = shape_gradient_wrt_ref_config_i * F_inv;
566  SymmetricTensor<2,dim> sym_shape_gradient_wrt_spt_config_i =
567  symmetrize(shape_gradient_wrt_spt_config_i);
568 
569  cell_rhs(i)-= (sym_shape_gradient_wrt_spt_config_i * Kirchhoffstress) * JxW;
570  //END - INSERT YOUR CODE HERE
571 
572  for(unsigned int j=0; j<dofs_per_cell; ++j)
573  {
574  //Assemble tangent contribution
575  //BEGIN - INSERT YOUR CODE HERE
576 
577  //Compute here the following
578  //
579  //- the symmetric (SymmetricTensor<2,dim>) gradient with respect to the spatial configuration
580  // for the ansatz function as discussed in the exercise (j) -> use fe_values_ref[u_fe] and F_inv
581  //- write into cell_matrix(i,j) the material and geometrical contributions for the tangent
582  Tensor<2,dim> shape_gradient_wrt_ref_config_j = fe_values_ref[u_fe].gradient(j,k);
583  Tensor<2,dim> shape_gradient_wrt_spt_config_j =shape_gradient_wrt_ref_config_j * F_inv;
584  SymmetricTensor<2,dim> sym_shape_gradient_wrt_spt_config_j =
585  symmetrize(shape_gradient_wrt_spt_config_j);
586 
587  cell_matrix(i,j) += (( symmetrize (transpose(shape_gradient_wrt_spt_config_i) *
588  shape_gradient_wrt_spt_config_j) * Kirchhoffstress ) //geometrical contribution
589  + (sym_shape_gradient_wrt_spt_config_i * Tangent // The material contribution:
590  * sym_shape_gradient_wrt_spt_config_j) )
591  * JxW;
592  //END - INSERT YOUR CODE HERE
593  }
594  }
595  }
596 
597 
598 
599  //Check for Neumann boundary condition
600  for(unsigned int face=0; face < GeometryInfo<dim>::faces_per_cell; ++face)
601  {
602  if(cell->face(face)->at_boundary() && cell->face(face)->boundary_id() == id_Neumann_boundary )
603  {
604  fe_face_values_ref.reinit(cell, face);
605 
606  for(unsigned int f_q_point = 0; f_q_point < n_q_points_f; ++f_q_point)
607  {
608  double step_fraction = double(current_load_step)/double(load_steps);
609  double current_load = load_magnitude * step_fraction;
610  //Compute the following
611  //
612  //- The normal vector of the current face at the current quadrature point (fe_face_values_ref)
613  //- The normal vector scaled with "current_load"
614  const Tensor<1,dim> NormalVector = fe_face_values_ref.normal_vector(f_q_point);
615  const Tensor<1,dim> Traction = current_load * NormalVector;
616 
617  for(unsigned int i = 0; i< dofs_per_cell; ++i)
618  {
619  //Compute
620  //
621  //- the test function at the face quadrature point for the i-th test function
622  //- write into cell_rhs(i)+= the contribution due to the Neumann boundary condition (-=(-))->+=
624  const Tensor<1,dim> shape_function = fe_face_values_ref[u_fe].value(i,f_q_point);
625  cell_rhs(i)+= (shape_function * Traction) * fe_face_values_ref.JxW(f_q_point);
626  }
627  }
628  }
629  }
630  //copy local to global
631  constraints.distribute_local_to_global(cell_matrix,cell_rhs,
632  local_dof_indices,
633  tangent_matrix,system_rhs,false);
634

639 template <int dim>

641 Solid<dim>::solve_linear_system(Vector<double> &newton_update)
642 {
643 
644  unsigned int lin_it = 0;
645  double lin_res = 0.0;
646  std::string solver_type = "CG";
647  /*reset the vector newton update*/
648  newton_update=0;
649 
650 
651  std::cout << " SLV " << std::flush;
652  if (solver_type == "CG")
653  {
654  const int solver_its = tangent_matrix.m()
655  * multiplier_max_iterations_linear_solver;
656  const double tol_sol = 1e-9
657  * system_rhs.l2_norm();
658 
659  SolverControl solver_control(solver_its, tol_sol);
660 
661  GrowingVectorMemory<Vector<double> > GVM;
662  SolverCG<Vector<double> > solver_CG(solver_control, GVM);
663  PreconditionSSOR<> preconditioner;
664  preconditioner.initialize(tangent_matrix, 1.2);
665  solver_CG.solve(tangent_matrix,
666  newton_update,
667  system_rhs,
668  preconditioner);
669  lin_it = solver_control.last_step();
670  lin_res = solver_control.last_value();
671  }
672  else
673  {
674  /*throug an error message that the chosen solver type is not implented*/
675  Assert (false, ExcMessage("Linear solver type not implemented"));
